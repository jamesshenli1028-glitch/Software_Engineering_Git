        +stopVacuum()

    }

    class SealingManager {
        +startSealing()
        +stopSealing()
    }

    SystemManager --> SafetyMonitor : uses
    SystemManager --> VacuumManager : controls
    SystemManager --> SealingManager : controls
    SafetyMonitor --> UIManager : triggers alarm

Aktivitätsdiagramm – Start/Stop

flowchart TD
    Start([Start Working Cycle])
    Check[Check Critical Components]
    OK{Components OK?}
    Run([Set status RUNNING])
    Vac([Start Vacuum])
    Seal([Start Sealing])
    End([Cycle Finished])

    Start --> Check
    Check --> OK
    OK -->|yes| Run
    Run --> Vac
    Vac --> Seal
    Seal --> End

    OK -->|no| Error([ERROR + Stop])


Aktivitätsdiagramm – Safety Monitoring

flowchart TD
    A([Monitor Loop])
    T{Temp >82°C?}
    C{Cover open?}
    M{Motor block?}

    A --> T
    T -->|yes| Shutdown([Notabschaltung + ERROR])
    T -->|no| C
    C -->|yes| Alarm([LED Rot + >70dB Alarm])
    C -->|no| M
    M -->|yes| Shutdown
    M -->|no| A
